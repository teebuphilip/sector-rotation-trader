from datetime import date
import os
import pandas as pd
from crazy.algos.base import CrazyAlgoBase
from crazy.utils import cached_fetch
from crazy.config import CRAZY_CACHE_DIR

class ExampleAlgo(CrazyAlgoBase):
    algo_id = "example"
    name = "Example"
    rebalance_frequency = "daily"
    supports_historical_seed = True

    def universe(self):
        return ["XLY", "XLK"]

    def _fetch_df(self):
        cache_path = os.path.join(CRAZY_CACHE_DIR, "example.json")

        def _fetch():
            # return list[dict] with a "date" field
            return []

        data = cached_fetch(cache_path, ttl_hours=24, fetch_fn=_fetch)
        return pd.DataFrame(data) if data else pd.DataFrame()

    def compute_signal(self, as_of: date, state: dict, historical: bool = False):
        df = self._fetch_df()
        if df.empty:
            return "HOLD"
        df = df[df["date"] <= pd.Timestamp(as_of)]
        if len(df) < 7:
            return "HOLD"
        df = df.copy()
        df["ma7"] = df["value"].rolling(7).mean()
        latest = df.iloc[-1]
        if latest["value"] > 1.5 * latest["ma7"]:
            return "RISK_ON"
        return "HOLD"

from datetime import date
import pandas as pd
from normal.algos.base import NormalAlgoBase
from config import SECTOR_ETFS

class ExampleNormalAlgo(NormalAlgoBase):
    algo_id = "example-normal"
    name = "Example Normal"
    rebalance_frequency = "monthly_eom"

    def universe(self):
        return list(SECTOR_ETFS)

    def compute_target(self, prices: pd.DataFrame, as_of: date):
        monthly = prices.resample("ME").last().dropna(how="all")
        if len(monthly) < 4:
            return {}
        top = monthly.iloc[-1].sort_values(ascending=False).head(3).index.tolist()
        weight = 1.0 / len(top)
        return {t: weight for t in top}
